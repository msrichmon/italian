  // Italian accents
  const IT_ACCENTS = ['à','è','é','ì','î','ò','ù','ü'];

  let cards = allCards.slice();
  let current = 0;
  let mode = 'it';
  let scores = {};
  let answered = false;
  let reviewMode = false;
  let originalCards = [];
  let skippedCards = {};

  function normalise(s) {
    return s.trim().toLowerCase().replace(/['']/g, "'").replace(/\s+/g, ' ');
  }

  function stripAccents(s) {
    return s.normalize('NFD').replace(/[\u0300-\u036f]/g, '');
  }

  function isCorrect(userInput, target) {
    const u = normalise(userInput);
    let t = normalise(target);
    // Strip parenthetical notes
    t = t.replace(/\s*\([^)]*\)/g, '');
    if (u === t) return true;
    // Multi-value answers separated by comma
    if (t.includes(',')) {
      const options = t.split(',').map(function(o) { return o.trim(); });
      for (var i = 0; i < options.length; i++) {
        if (u === options[i]) return true;
      }
    }
    if (u.includes(',')) {
      const uOpts = u.split(',').map(function(o) { return o.trim(); });
      const tOpts = t.split(',').map(function(o) { return o.trim(); });
      var allValid = true;
      for (var i = 0; i < uOpts.length; i++) {
        var found = false;
        for (var j = 0; j < tOpts.length; j++) {
          if (uOpts[i] === tOpts[j]) { found = true; break; }
        }
        if (!found) { allValid = false; break; }
      }
      if (allValid && uOpts.length > 0) return true;
    }
    if (mode === 'it') {
      // Accept without "the"
      const tNoThe = t.replace(/^the\s+/, '');
      if (u === tNoThe) return true;
    }
    if (mode === 'en') {
      // Accept with or without accents
      if (stripAccents(u) === stripAccents(t)) return true;
      // Accept without Italian articles
      const tNoArt = t.replace(/^(il|lo|la|l'|i|gli|le)\s*/i, '');
      if (u === normalise(tNoArt) || stripAccents(u) === stripAccents(tNoArt)) return true;
    }
    return false;
  }

  function renderCard() {
    const card = cards[current];
    if (!card) return;
    answered = false;

    if (mode === 'it') {
      document.getElementById('frontLabel').textContent = 'ITALIAN';
      document.getElementById('frontWord').textContent = card.it;
    } else {
      document.getElementById('frontLabel').textContent = 'ENGLISH';
      document.getElementById('frontWord').textContent = card.en;
    }

    const exampleEl = document.getElementById('cardExample');
    if (card.example && card.exampleEn) {
      exampleEl.textContent = mode === 'it' ? card.example : card.exampleEn;
      exampleEl.style.display = 'block';
    } else {
      exampleEl.textContent = '';
      exampleEl.style.display = 'none';
    }

    const input = document.getElementById('answerInput');
    input.value = '';
    input.disabled = false;
    input.className = 'answer-input';
    document.getElementById('checkBtn').disabled = false;
    document.getElementById('resultOverlay').className = 'result-overlay';
    document.getElementById('feedbackStrip').className = 'feedback-strip';
    document.getElementById('nextBtn').classList.remove('show');

    renderAccentBar();

    const pct = ((current + 1) / cards.length) * 100;
    document.getElementById('progressFill').style.width = pct + '%';
    document.getElementById('progressText').textContent = 'Card ' + (current + 1) + ' of ' + cards.length;

    setTimeout(function() { document.getElementById('answerInput').focus(); }, 100);
  }

  function renderAccentBar() {
    const bar = document.getElementById('accentBar');
    if (mode !== 'en') { bar.innerHTML = ''; return; }
    bar.innerHTML = IT_ACCENTS.map(function(a) {
      return '<button class="accent-btn" data-char="' + a + '">' + a + '</button>';
    }).join('');
    var btns = bar.querySelectorAll('.accent-btn');
    for (var i = 0; i < btns.length; i++) {
      btns[i].onclick = function() { insertAccent(this.getAttribute('data-char')); };
    }
  }

  function insertAccent(char) {
    const input = document.getElementById('answerInput');
    if (input.disabled) return;
    const s = input.selectionStart, e = input.selectionEnd;
    input.value = input.value.slice(0, s) + char + input.value.slice(e);
    input.selectionStart = input.selectionEnd = s + 1;
    input.focus();
  }

  function checkAnswer() {
    if (answered) { nextCard(); return; }
    const card = cards[current];
    const input = document.getElementById('answerInput');
    const userValue = input.value;
    if (!userValue.trim()) return;

    const target = mode === 'it' ? card.en : card.it;
    const correct = isCorrect(userValue, target);

    answered = true;
    input.disabled = true;
    document.getElementById('checkBtn').disabled = true;
    delete skippedCards[current];

    const overlay = document.getElementById('resultOverlay');
    const fb = document.getElementById('feedbackStrip');
    const resultExampleEl = document.getElementById('resultExample');

    if (correct) {
      input.classList.add('correct-input');
      overlay.className = 'result-overlay correct-overlay show';
      document.getElementById('resultIcon').textContent = '✓';
      document.getElementById('resultMsg').textContent = 'Corretto! 🎉';
      document.getElementById('resultAnswer').textContent = target;
      fb.textContent = '✓  "' + target + '"';
      fb.className = 'feedback-strip correct-strip show';
      scores[current] = true;
    } else {
      input.classList.add('wrong-input');
      overlay.className = 'result-overlay wrong-overlay show';
      document.getElementById('resultIcon').textContent = '✗';
      document.getElementById('resultMsg').textContent = 'Non proprio…';
      document.getElementById('resultAnswer').textContent = 'Answer: ' + target;
      fb.textContent = '✗  The answer was: "' + target + '"';
      fb.className = 'feedback-strip wrong-strip show';
      scores[current] = false;
    }

    if (card.exampleEn && card.example) {
      resultExampleEl.textContent = mode === 'it' ? card.exampleEn : card.example;
      resultExampleEl.style.display = 'block';
    } else {
      resultExampleEl.textContent = '';
      resultExampleEl.style.display = 'none';
    }

    updateStats();
    document.getElementById('nextBtn').classList.add('show');
  }

  function nextCard() {
    if (current < cards.length - 1) {
      current++;
      renderCard();
    } else {
      const correct = Object.keys(scores).filter(function(k) { return scores[k] === true; }).length;
      const fb = document.getElementById('feedbackStrip');
      fb.textContent = '🎉 Completato! ' + correct + '/' + cards.length + ' correct. Shuffle to go again!';
      fb.className = 'feedback-strip correct-strip show';
      document.getElementById('nextBtn').classList.remove('show');
    }
  }

  function skipCard() {
    skippedCards[current] = true;
    if (current < cards.length - 1) { current++; renderCard(); }
    updateStats();
  }

  function updateStats() {
    var correct = Object.keys(scores).filter(function(k) { return scores[k] === true; }).length;
    var incorrect = Object.keys(scores).filter(function(k) { return scores[k] === false; }).length;
    var skipped = Object.keys(skippedCards).length;
    var remaining = Math.max(0, cards.length - Object.keys(scores).length - skipped);
    document.getElementById('statCorrect').textContent = correct;
    document.getElementById('statIncorrect').textContent = incorrect;
    document.getElementById('statRemaining').textContent = remaining;
    document.getElementById('reviewBtn').disabled = incorrect === 0;
    document.getElementById('reviewSkippedBtn').disabled = skipped === 0;
  }

  function setCategory(cat) {
    ['all','viso','corpo','prepositions','conjugations'].forEach(function(c) {
      document.getElementById('tab-' + c).classList.toggle('active', c === cat);
    });
    cards = cat === 'all' ? allCards.slice() : allCards.filter(function(c) { return c.category === cat; });
    scores = {}; skippedCards = {}; current = 0; reviewMode = false;
    document.getElementById('reviewBanner').classList.remove('show');
    renderCard(); updateStats();
  }

  function setMode(m) {
    mode = m;
    document.getElementById('mode-it').classList.toggle('active', m === 'it');
    document.getElementById('mode-en').classList.toggle('active', m === 'en');
    scores = {}; skippedCards = {}; current = 0; reviewMode = false;
    document.getElementById('reviewBanner').classList.remove('show');
    renderCard(); updateStats();
  }

  function shuffleCards() {
    if (reviewMode) {
      cards = originalCards.slice(); reviewMode = false; originalCards = [];
      document.getElementById('reviewBanner').classList.remove('show');
    }
    for (var i = cards.length - 1; i > 0; i--) {
      var j = Math.floor(Math.random() * (i + 1));
      var t = cards[i]; cards[i] = cards[j]; cards[j] = t;
    }
    scores = {}; skippedCards = {}; current = 0;
    renderCard(); updateStats();
  }

  function reviewMissed() {
    var missed = Object.keys(scores).filter(function(k) { return scores[k] === false; }).map(Number);
    if (missed.length === 0) return;
    originalCards = cards.slice(); reviewMode = true;
    cards = missed.map(function(i) { return originalCards[i]; });
    scores = {}; current = 0;
    document.getElementById('reviewBanner').innerHTML = '🔁 Reviewing missed words only <button id="exitReviewBtn">Exit Review</button>';
    document.getElementById('exitReviewBtn').onclick = exitReview;
    document.getElementById('reviewBanner').classList.add('show');
    renderCard(); updateStats();
  }

  function reviewSkipped() {
    var skippedIdx = Object.keys(skippedCards).filter(function(k) { return skippedCards[k]; }).map(Number);
    if (skippedIdx.length === 0) return;
    originalCards = cards.slice(); reviewMode = true;
    cards = skippedIdx.map(function(i) { return originalCards[i]; });
    skippedCards = {}; scores = {}; current = 0;
    document.getElementById('reviewBanner').innerHTML = '🔁 Reviewing skipped words only <button id="exitReviewBtn">Exit Review</button>';
    document.getElementById('exitReviewBtn').onclick = exitReview;
    document.getElementById('reviewBanner').classList.add('show');
    renderCard(); updateStats();
  }

  function exitReview() {
    if (!reviewMode) return;
    cards = originalCards.slice(); reviewMode = false; originalCards = []; current = 0;
    document.getElementById('reviewBanner').classList.remove('show');
    renderCard(); updateStats();
  }

  // Event bindings
  document.getElementById('checkBtn').onclick = function(e) { e.preventDefault(); checkAnswer(); };
  document.getElementById('nextBtn').onclick = function(e) { e.preventDefault(); nextCard(); };
  document.getElementById('skipBtn').onclick = function(e) { e.preventDefault(); skipCard(); };
  document.getElementById('shuffleBtn').onclick = function(e) { e.preventDefault(); shuffleCards(); };
  document.getElementById('reviewBtn').onclick = function(e) { e.preventDefault(); reviewMissed(); };
  document.getElementById('reviewSkippedBtn').onclick = function(e) { e.preventDefault(); reviewSkipped(); };
  document.getElementById('exitReviewBtn').onclick = function(e) { e.preventDefault(); exitReview(); };

  document.getElementById('tab-all').onclick = function() { setCategory('all'); };
  document.getElementById('tab-viso').onclick = function() { setCategory('viso'); };
  document.getElementById('tab-corpo').onclick = function() { setCategory('corpo'); };
  document.getElementById('tab-prepositions').onclick = function() { setCategory('prepositions'); };
  document.getElementById('tab-conjugations').onclick = function() { setCategory('conjugations'); };

  document.getElementById('mode-it').onclick = function() { setMode('it'); };
  document.getElementById('mode-en').onclick = function() { setMode('en'); };

  document.getElementById('answerInput').onkeypress = function(e) {
    if (e.key === 'Enter' || e.keyCode === 13) {
      e.preventDefault();
      answered ? nextCard() : checkAnswer();
    }
  };

  document.addEventListener('keydown', function(e) {
    if ((e.key === 'Enter' || e.keyCode === 13) && answered && document.getElementById('nextBtn').classList.contains('show')) {
      e.preventDefault(); nextCard();
    }
  });

  renderCard();
  updateStats();
})();
